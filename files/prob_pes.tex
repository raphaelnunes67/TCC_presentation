\begin{frame}{Problema de Pesquisa}
\begin{block}{Sistemas inertes}
\justify{Sistemas de iluminação e de medição de energia elétrica prediais ainda não dispõem de recursos tecnológicos que melhorem a eficiência energética e permitam a interoperabilidade com o usuário.}
\end{block}
\begin{block}{Custo-benefício}
\justify{Existem poucas soluções práticas e de baixo custo para realizar transições tecnológicas em sistemas passivos.}
\end{block}
\end{frame}
